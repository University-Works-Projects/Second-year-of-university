SOLUZIONE:
codice equivalente e commentato:

if case:
    int X = 1
    int Y
    void fie()
        foo()           // 3.
        X = 0           // 5. X=0 (non fa riferimento alla X di foo() xke quell'ambiente è stato oramai
                                   creato e distrutto. Fa riferimento alla X di 1.)
    void foo()
        int X = 5       // 4. X=5 (è una X locale, non è la X di 1.)

    input(Y)            // 0. Y=5
    if Y > 0 then
        int X = 4       // 1. X=4
        fie()           // 2.
    else
        fie()
    wrtie(x)            // 6. print: 1 (ovvero la X globale. La X di 1. è distrutta dopo il concludersi
                                        del blocco dell'if)

else case:
    int X = 1
    int Y
    void fie()
        foo()           // 2.
        X = 0           // 4. X=0 (si fa riferimento alla X globale, quella di foo() è distrutta)
    void foo()
        int X = 5       // 3. X=5

    input(Y)            // 0. Y=-3
    if Y > 0 then
        int X = 4
        fie()
    else
        fie()           // 1.
    wrtie(x)            // 5. print: 0 (ovvero la X globale)
