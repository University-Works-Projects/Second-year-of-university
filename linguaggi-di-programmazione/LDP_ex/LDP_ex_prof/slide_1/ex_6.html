SOLUZIONE:
codice equivalente e commentato:

int x=3, y=4
void foo(int w, int z)      // call(3,3)
    int k = 5
    w = w + 1               // w=3+1
    if (z == w)             // 3 == 4 -> False: else case
        write(k)
    else
        write(w)            // print: 4

foo(x,x)
write(x)                    // print: 3
write(y)                    // print: 4
