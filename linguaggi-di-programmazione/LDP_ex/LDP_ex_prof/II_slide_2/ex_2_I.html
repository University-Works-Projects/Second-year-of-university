SOLUZIONE:
codice equivalente ed esecuzione commentata:

int x=3
int y=4
void foo(reference int w, reference int z)      // foo(3,3)
    int k=5
    w=w+1           // w=3+1 -> z=4
    if z==w         // if 4==4
        write(k)    // print: 5
    else write(w)
foo(x,x)
write(x)    // print: 4
write(y)    // print: 4
