/* Explanation */
void foo ( int* x, int* y, int* j ){        // Call: foo(x(that points to x[0]), &x[1]==2, &i==1)
    x[ *j ] = *j + 1;                       // x[0+1]=1+1=2
    print( *y );                            // print: 2 (cuz &x[1]==2)
    x++;                                    // x now points to x[1]
    x[ *j ] = *j;                           // x[1+1]=1 (1+1 per via di "x++;")
    print( x[ *j ], *j );                   // print: (x[1+1]==)1, 1
    *j = *j + 1;                            // &i=1+1=2
}

int[] x[ 10 ];
int i = 1;
x[ 0 ] = 1;
x[ 1 ] = 2;
x[ 2 ] = 3;
foo( x, &x[ i ], &i );
print( x[ i ], i );                         // print: 1 (cuz x points to x[0], so x[0+2]==1) 2 (cuz i==2)
