Scope: statico; Passaggio: per valore

int x = 3

void foo(int y)
    int x = 5
    x = x + y
    x = x + y
    write(x)
    write(y)

foo(x++)
write(x)

SOLUZIONE:
codice equivalente e commentato:

int x = 3
void foo(int y)     // call: foo(3++)
    int k = 5
    k = k + y       // k=5+3
    k = k + y       // k=8+3
    write(k)        // print: 11
    write(y)        // print: 3
foo(x++)
write(x)            // print: 4
