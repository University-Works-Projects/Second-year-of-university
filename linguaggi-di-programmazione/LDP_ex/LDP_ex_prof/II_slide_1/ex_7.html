SOLUZIONE:
codice commentato:

int x = 4
void foo(int y)     // foo(10) -> y=10
    int w           // 2.
    x = x + y       // 3. x=10+10   (si fa riferimento alla x di 0.)
    w = y           // 4. w=10
    write(w)        // 5. print: 10
    write(y)        // 6. print: 10
{
    int x = 10      // 0.
    foo(x)          // 1.
    write(x)        // 7. print: 20
}
