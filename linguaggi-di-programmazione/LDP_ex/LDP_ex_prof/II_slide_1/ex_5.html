Scope: statico; Passaggio: per riferimento

int x = 2
void foo(reference int y)
    x = x + 1
    y = y + 10
    x = x + y
    write(x)

{
    int x = 10
    foo(x)
    write(x)
}


SOLUZIONE:
codice equivalente e commentato:

int x = 2
void foo(reference int y)       // call: foo(10)
    x = x + 1                   // x=2+1
    y = y + 10                  // &k=10+10
    x = x + y                   // x=3+20
    write(x)                    // print: 23
{
    int k = 10
    foo(k)
    write(k)                    // print: 20
}
